fn malloc(size: u64) -> *i8;

struct Human {
    name: *i8;
    age: i32;
};

impl Human {
    fn new(name: *i8, age: i32) -> Human {
        let mut self: Human = [name, age];
        return self;
    }
}

fn main(argc: i32, argv: **i8) -> i32 {
    let max = Human::new("Max", 18);
    printf_ln("%s -> %d", max.name, max.age);
    return 0;
}
