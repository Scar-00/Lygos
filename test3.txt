struct Foo {
    x: i32;
};

impl Foo {
    fn test(&mut self) {
        printf("%d\n", (*self).x);
        printf("YESSSS\n");
    }
}

struct Human {
    name: *i8;
    age: i32;
    foo: Foo;
};

impl Human {
    fn new(name: *i8, age: i32) -> Human {
        let mut this: Human;
        this.name = name;
        this.age = age;
        let mut foo: Foo;
        foo.x = 1000;
        this.foo = foo;
        return this;
    }

    fn test(&mut self, x: i32) -> Foo {
        return (*self).foo;
    }
}

fn test(x: i32) -> i32 {
    return 10;
}

fn main(argc: i32, argv: **i8) -> i32 {
    let max = Human::new("Max", 18);
    max.foo.test();
    printf("%s -> %d\n", max.name, max.age);
    return 0;
}
