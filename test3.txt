fn malloc(size: u64) -> *i8;

struct Foo {
    x: i32;
};

impl Foo {
    fn test(&mut self) {
        printf("%d\n", self->x);
        printf("YESSSS\n");
    }
}

struct Human {
    name: *i8;
    age: i32;
    foo: Foo;
};

impl Human {
    fn new(name: *i8, age: i32) -> Human {
        let mut this: Human;
        this.name = name;
        this.age = age;
        let mut foo: Foo;
        foo.x = 1000;
        this.foo = foo;
        return this;
    }

    fn test(&mut self, x: i32) -> Foo {
        return (*self).foo;
    }
}

fn test() {
    for let y = 0 in y < 10 {
        for let x = 0 in x < 10 {
            if y == 0 {
                printf("# ");
            }else {
                printf("* ");
            }
            x = x + 1;
        }
        printf("\n");
        y = y + 1;
    }
}

fn main(argc: i32, argv: **i8) -> i32 {
    let mut foo: Foo;
    foo.x = 100;
    foo.test();
    return 0;
}
