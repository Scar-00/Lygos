fn malloc(size: i32) -> *i32;
fn print_deref(ptr: *i32);

struct Foo {
    x: i32;
};

fn make_foo() -> Foo {
    let mut x: Foo;
    x.x = 9;
    return x;
}

fn take_ref(x: i32) -> *i32 {
    let y = &x;
    let p = y;
    return p;
}

fn deref(x: *i32) -> i32 {
    let mut y: i32;
    y = *x;
    return y;
}

fn main() -> i32 {
    let foo = make_foo();
    let x = deref(&foo.x);
    return x;
}
